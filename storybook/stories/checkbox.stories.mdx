import { Meta, Story, Canvas, ArgsTable } from '@storybook/addon-docs';
import { CustomElement, PackageInfo } from './Components';

import '@equinor/fusion-wc-checkbox';

export const Template = (args) => (
  <div>
    <h5>Standard</h5>
    <div style={{display: 'flex', gap: 5}}>
      <fwc-checkbox {...args} />
      <fwc-checkbox {...args} checked/>
      <fwc-checkbox {...args} indeterminate/>
    </div>
    <h5>Disabled</h5>
    <div style={{display: 'flex', gap: 5}}>
      <fwc-checkbox {...args} disabled/>
      <fwc-checkbox {...args} disabled checked/>
      <fwc-checkbox {...args} disabled indeterminate/>
    </div>
  </div>
);

<Meta title="Input/Checkbox" />

<PackageInfo pkg={require('@equinor/fusion-wc-checkbox/package.json')}/>

<CustomElement tag="fwc-checkbox" data={require('@equinor/fusion-wc-checkbox/lib/custom-elements.json')} />

<Canvas>
  <Story 
    name="Basic"
    argTypes={{
      size: {
        type: 'number',
        name: '--fwc-checkbox-size',
        table: {
          category: 'style'
        }
      },
      spacing: {
        type: 'number',
        name: '--fwc-checkbox-spacing',
        table: {
          category: 'style'
        }
      }
    }}
    args={{
      checked: false,
      reducedTouchTarget: false,
    }}
  >
    {({checked, reducedTouchTarget, size, spacing}) => {
      const props = {};
      checked && (props.checked = true);
      reducedTouchTarget && (props.reducedTouchTarget = true);
      
      const style = {}
      size && (style['--fwc-checkbox-size'] = size+'px');
      spacing && (style['--fwc-checkbox-spacing'] = spacing+'px');
      
      return <fwc-checkbox {...props} style={style}/>;
    }}
  </Story>
</Canvas>

<ArgsTable story="Basic" />
