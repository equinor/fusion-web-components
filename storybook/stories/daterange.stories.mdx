export const Relative = ({}) => {
  return html`<fwc-daterange variant="relative" from="2021-08-09T09:12:49Z" to="2021-08-09T11:12:49Z" />`;
};

export const Distance = ({}) => {
  return html`<fwc-daterange variant="distance" from="2021-08-09T09:12:49Z" to="2021-08-09T11:12:49Z" addSuffix />`;
};

export const DateTime = ({}) => {
  return html`<fwc-daterange
    variant="datetime"
    from="2021-08-09T09:12:49Z"
    to="2021-08-09T11:12:49Z"
    format=${DateTimeFormat.date}
  />`;
};

export const DateTimeSeparator = ({}) => {
  return html`<fwc-daterange
    variant="datetime"
    from="2021-08-09T09:12:49Z"
    to="2021-08-09T11:12:49Z"
    format=${DateTimeFormat.date}
  >
    <span slot="separator">to</span>
  </fwc-daterange>`;
};

export const Locale = ({}) => {
  return html`<fwc-daterange variant="relative" from="2021-08-09T09:12:49Z" to="2021-08-09T11:12:49Z" locale="nb" />`;
};
