  <div>
    <h5>Standard</h5>
    <div style={{display: 'flex', gap: 5}}>
      <fwc-radio {...args} />
      <fwc-radio {...args} checked/>
      <fwc-radio {...args} indeterminate/>
    </div>
    <h5>Disabled</h5>
    <div style={{display: 'flex', gap: 5}}>
      <fwc-radio {...args} disabled/>
      <fwc-radio {...args} disabled checked/>
      <fwc-radio {...args} disabled indeterminate/>
    </div>
  </div>
);

<Meta title="Input/Radio" />

<PackageInfo pkg={require('@equinor/fusion-wc-radio/package.json')}/>

<CustomElement tag="fwc-radio" data={require('@equinor/fusion-wc-radio/lib/custom-elements.json')} />

<Canvas>
  <Story 
    name="Basic"
    argTypes={{
      size: {
        type: 'number',
        name: '--fwc-radio-size',
        table: {
          category: 'style'
        }
      },
      spacing: {
        type: 'number',
        name: '--fwc-radio-spacing',
        table: {
          category: 'style'
        }
      },
      color: {
        control: 'color',
        name: '--fwc-radio-color',
        table: {
          category: 'style'
        }
      }
    }}
    args={{
      checked: false,
      reducedTouchTarget: false,
    }}
  >
    {({checked, reducedTouchTarget, size, spacing, color}) => {
      const props = {};
      checked && (props.checked = true);
      reducedTouchTarget && (props.reducedTouchTarget = true);
      
      const style = {}
      size && (style['--fwc-radio-size'] = size+'px');
      spacing && (style['--fwc-radio-spacing'] = spacing+'px');
      color && (style['--fwc-radio-color'] = color);
      
      return <fwc-radio {...props} style={style}/>;
    }}
  </Story>
</Canvas>

<ArgsTable story="Basic" />
